softmax = torch.nn.Softmax(dim=-1)
class Node:
    def __init__(self, parent=None, move=None):
        self.value_sum = 0
        self.visit_count = 0
        self.predicted_probs = None
        self.parent = parent
        self.move = move
        self.children = None
        self.children_moves = None

    def get_expected_value(self):
        return self.value_sum / self.visit_count if self.visit_count else 0

    def is_root(self):
        return self.parent is None
    
    def expand(self, state):
        if self.children is not None or not list(state['board'].legal_moves):
            return
        self.children_moves = [move.uci() for move in state['board'].legal_moves]
        self.children = {move: Node(parent=self, move=move) for move in self.children_moves}
        self.prior = CNN.predict(state)[0]
    
    def upper_confidence_bound_score(self, prior):
        return self.get_expected_value() + (prior * np.sqrt(np.log(self.parent.visit_count+1) / (self.visit_count+1)))
    
    def upper_confidence_bound_score_children(self):
        return torch.from_numpy(np.fromiter(map(lambda x: self.children[x[0]].upper_confidence_bound_score(x[1]), zip(self.children_moves, self.prior)), dtype=np.float64))
    
    def get_next_move(self):
        ucb_score_children = self.upper_confidence_bound_score_children()
        noisy_ucb_score_children = ucb_score_children + np.random.normal(scale=max(max(ucb_score_children)/10000,0.00001), size=len(ucb_score_children))
        # Adds random noise to break ties
        argmax = np.argmax(noisy_ucb_score_children)
        return self.children_moves[argmax]
    
    def get_probability_distribution(self):
        return softmax(torch.from_numpy(np.fromiter(map(lambda x: self.children[x].visit_count, self.children_moves), dtype=np.float32))).type(torch.float32)

class CNN(torch.nn.Module):
    def __init__(self):
        super(CNN, self).__init__()
        self.board_mask_embedding = torch.nn.Embedding(BOARD_EMBEDDING_SIZE, BOARD_EMBEDDING_DIM)
        self.conv_1 = torch.nn.Conv2d(NUM_INPUT_CHANNELS, *CONV_1, device=device)
        self.conv_2 = torch.nn.Conv2d(CONV_1[0], *CONV_2, device=device)
        self.flatten = torch.nn.Flatten(start_dim=0)
#         self.dropout = torch.nn.Dropout(p=DROPOUT_P)
        self.dense_1 = torch.nn.LazyLinear(DENSE_1)
        self.dense_2 = torch.nn.LazyLinear(DENSE_2)
        self.dense_3 = torch.nn.LazyLinear(DENSE_3)
        self.dense_4 = torch.nn.LazyLinear(DENSE_4)
        self.dense_5 = torch.nn.LazyLinear(DENSE_5)
        self.relu = torch.nn.ReLU(inplace=False)
        self.tanh = torch.nn.Tanh()
        self.softmax = torch.nn.Softmax(dim=-1)

        
    def forward(self, x):
        output = x
        output = self.conv_1(output)
        output = self.conv_2(output)
        output = self.flatten(output)
        
        output_iter = self.dense_1(output)
        output_iter = self.tanh(output_iter)
        
#         ev = self.dropout(output_iter)
        ev = self.dense_4(output_iter)
        ev = self.relu(ev)
#         ev = self.dropout(ev)
        ev = self.dense_5(ev)
        ev = self.tanh(ev)
        
#         output = self.dropout(output_iter)
        output = self.dense_2(output)
        output = self.tanh(output)
        
#         output = self.dropout(output)
        output = self.dense_3(torch.cat([output, output_iter], axis=-1))
        
        return output.reshape((8, 8, -1)), ev


    def predict(self, state):
        stack_name = ('black_stack', 'white_stack')[state['color']]
        stack_slice = state[stack_name][-T:]
        constant_params = state['constant_stack'][-1]
        
        board_mask_history_embedding = self.board_mask_embedding(torch.stack(stack_slice, axis=0))
        # padding board history
        board_mask_history_embedding = torch.cat([torch.zeros((T-board_mask_history_embedding.shape[0], 8, 8, BOARD_EMBEDDING_DIM)), board_mask_history_embedding], axis=0)
        board_mask_history_embedding = board_mask_history_embedding.permute([1, 2, 3, 0]).reshape([8, 8, -1])

        constant_state = torch.tensor(constant_params)
        constant_state = (torch.ones((8, 8, 8), dtype=torch.float32) * constant_state.unsqueeze(axis=0).unsqueeze(axis=0))
        board_state = torch.cat([board_mask_history_embedding, constant_state], axis=2).permute([2, 0, 1]).to(device)

        pred, ev = self.forward(board_state)
        moves = [x.uci() for x in state['board'].legal_moves]
        encodings = [encode_from_uci(move, white=state['color']) for move in moves]
        pred = torch.stack([pred[encoding] for encoding in encodings], axis=0)
        with torch.no_grad():
            softmax_pred = self.softmax(pred)
            next_move = moves[Categorical(softmax_pred).sample()]
        return pred, ev, next_move

square_dict = {col_ch + row_ch: np.array([col, row]) for row, row_ch in enumerate("12345678") for col, col_ch in enumerate("abcdefgh")}
def encode_from_uci(string, white=True):
    start, end, promotion = string[:2], string[2:4], None if len(string) == 4 else string[-1]
    start, end = square_dict[start], square_dict[end]
    if not white:
        start, end = 7 - start, 7 - end
    
    delta = tuple(end - start)
    if promotion is not None and promotion != 'q':
        delta = delta + (promotion,)
    return tuple(start) + (delta_mapping[delta],)

def update_board_mask(board_mask, move, color=True):
    board_mask = torch.clone(board_mask)
    start, end, promotion = move[:2], move[2:4], move[4:] if move[4:] else None
    start, end = square_dict_board_update[start], square_dict_board_update[end]
    board_mask[start], board_mask[end] = 0, board_mask[start].item()
    if promotion is not None:
        piece = promotion_pieces[(promotion, color)]
        board_mask[end] = piece
    return board_mask, bool(promotion) or (board_mask[end] in (1, 12))

Game State Representation¶

6 elements:

- board

- repitition count

- white stack

- black stack (mirror of white stack)

- constant stack

- color (for ease of access)

After selecting the next move:

board and repitition count will be changed in place but the 3 stacks will have a new element appended at the end

Within the constant stack there are 8 variables:

def initialize_state():
    board = chess.Board()
    board_mask = torch.clone(start_board_mask)
    repitition_count = defaultdict(int)
    white_stack, black_stack = [board_mask], [flip_board(board_mask)]
    repitition_count[tensor_to_tuple(white_stack[-1])] += 1

    no_progress_count = repitition_count_current = 0
    color = move_count = p1_lc = p1_rc = p2_lc = p2_rc = 1

    constant_stack = [(move_count, color, no_progress_count, repitition_count_current, p1_lc, p1_rc, p2_lc, p2_rc)]
    return {"board": board, "repititon_count_dict": repitition_count, "white_stack": white_stack, "black_stack": black_stack, "constant_stack": constant_stack, "color": color}

def update_state(state, move):
    state['board'].push_uci(move)
    board_mask_white, progress = update_board_mask(state['white_stack'][-1], move, color=state['color']) # assumes color is the color of the moving player
    state['white_stack'].append(board_mask_white)
    state['black_stack'].append(flip_board(board_mask_white))
    repitition_count_current = state['repititon_count_dict'][tensor_to_tuple(board_mask_white)]
    state['repititon_count_dict'][tensor_to_tuple(board_mask_white)] += 1
    state['color'] = 1 - state['color']

    move_count, _, no_progress_count, _, _, _, _, _ = state['constant_stack'][-1]
    move_count += 1
    no_progress_count = no_progress_count + 1 if not progress else 0

    p1_lc, p1_rc, p2_lc, p2_rc = map(lambda x: int(bool(state['board'].castling_rights & x)), (chess.BB_A1, chess.BB_H1, chess.BB_H8, chess.BB_A8))
    if not state['color']:
        p1_lc, p1_rc, p2_lc, p2_rc = p2_lc, p2_rc, p1_lc, p1_rc
    state['constant_stack'].append((move_count, state['color'], no_progress_count, repitition_count_current, p1_lc, p1_rc, p2_lc, p2_rc))

def previous_state(state):
    state['board'].pop()
    state['repititon_count_dict'] = None # Can't go backward so avoiding misuse
    state['white_stack'].pop()
    state['black_stack'].pop()
    state['constant_stack'].pop()
    state['color'] = 1 - state['color']

def game_over(state, print_=False):
    if not print_:
        return state['board'].is_checkmate() or state['board'].is_stalemate() or state['board'].is_insufficient_material() or state['board'].can_claim_threefold_repetition()
    elif state['board'].is_checkmate():
        print("Checkmate")
    elif state['board'].is_insufficient_material():
        print("Insufficient Material")  
    elif state['board'].can_claim_threefold_repetition():
        print("Threefold Repitition")
    elif state['board'].is_stalemate():
        print("Other Stalemate")
    else:
        return False
    return True

def simulate_against_random_opponent(num_simulations=256, white=None, print_=False, display_moves=False):
    if white is None:
        white = random.getrandbits(1)
    draws = white_wins = black_wins = 0
    with torch.no_grad():
        for _ in range(num_simulations):
            state_ = initialize_state()        
            while not game_over(state_, print_=print_outcome):
                if state_['color'] == white:
                    next_move = (list(state_['board'].legal_moves)[np.argmax(CNN.predict(state_)[0])]).uci()
                else:
                    next_move = np.random.choice(list(state_['board'].legal_moves)).uci()
                update_state(state_, next_move)
                if display_moves:
                    display(state_['board'])
            if state_['board'].is_checkmate() and not state_['color']:
                white_wins += 1
            elif state_['board'].is_checkmate():
                black_wins += 1
            else:
                draws += 1
    return ((white << 1) - 1) * (white_wins - black_wins) / num_simulations

training_loss = []
liveloss = PlotLosses()
for epoch_outer in tqdm(range(num_epochs)):
    pred_game_length = np.percentile(previous_lengths, np.random.uniform(high=100))
    game_start_level = int(np.random.uniform() * pred_game_length)
    
    num_attempts = 0
    valid = False
    while not valid:
        state_original = initialize_state()
        for _ in range(game_start_level):
            if game_over(state_original):
                break
            next_move = np.random.choice(list(state_original['board'].legal_moves)).uci()
            update_state(state_original, next_move)
        valid = not game_over(state_original)
        num_attempts += 1

    root = Node()
    root.expand(state_original)
            

    for epoch_inner in range(epochs_per_start_position):
        ############## INITIALIZATION ############

        state = copy_state(state_original)

        ############## SELECTION ############

        node = root
        while node.children is not None:
            next_move = node.get_next_move()
            update_state(state, next_move)
            node = node.children[next_move]
        node.expand(state)

        ############## SIMULATION ############

        draws = white_wins = black_wins = 0
        with torch.no_grad():
            for _ in range(num_simulations):
                state_ = copy_state(state)
                while not game_over(state_, print_=print_outcome):
                    next_move = CNN.predict(state_)[-1]
                    update_state(state_, next_move)
                    if display_moves:
                        display(state_['board'])

                if state_['board'].is_checkmate() and not state_['color']:
                    white_wins += 1
                elif state_['board'].is_checkmate():
                    black_wins += 1
                else:
                    draws += 1
                previous_lengths.append(state_['constant_stack'][-1][0]-1)
#             print(f"White wins: {white_wins}\tBlack wins: {black_wins}\tDraws: {draws}")

        ########### BACKUP ##################
        expected_value_abs = (white_wins - black_wins) * 4
        while node != root:
            previous_state(state)
            node.value_sum += -(1-(2*state['color']))*(expected_value_abs)
            node.visit_count += 1
            node = node.parent
        root.value_sum -= expected_value_abs
        root.visit_count += 1

    pred, pred_ev, _ = CNN.predict(state)
    true_dist = node.get_probability_distribution()
    true_ev = torch.tensor(-node.get_expected_value()/(4 * num_simulations)).type(torch.float32)

    loss = criterion(pred, pred_ev, true_dist, true_ev)
    torch.nn.utils.clip_grad_norm(CNN.parameters(), 1)
    loss.backward()
    optimizer.step()

    current_loss = loss.item()
    training_loss.append(current_loss)
    logs = {}

#     print(f"Epoch {epoch_outer}\t\tLoss: {current_loss}")
    if not epoch_outer % validate_every:
        validation_accuracy = simulate_against_random_opponent(num_simulations=validation_simulations)

        logs['validation_accuracy'] = validation_accuracy
        if epoch_outer:
            assert len(training_loss) == validate_every
        logs['cross_entropy'] = np.mean(training_loss)
        training_loss = []

        liveloss.update(logs)
        liveloss.send()
